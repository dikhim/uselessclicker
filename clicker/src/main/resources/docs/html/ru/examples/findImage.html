function findDots(){

    stopTimer();
    system.println(p.x + ':' + p.y);

    startTimer();
    var p = parent.findFirst(greenDot);
    stopTimer();
    system.println(p.x + ':' + p.y);
    system.println('');
    
    system.println('find a first purple dot');
    startTimer();
    var p = parent.findFirst(purpleDot);
    stopTimer();
    system.println(p.x + ':' + p.y);
    system.println('');
    
    
    // findAll
    system.println('***** findAll *****');
    system.println('find all blue dots');
    startTimer();
    var list = parent.findAll(blueDot);
    stopTimer();
    system.println('found '+list.size());
    system.println('');
       
    system.println('find all red dots');
    startTimer();
    var list = parent.findAll(redDot);
    stopTimer();
    system.println('found '+list.size());
    system.println('');
    
    // findLimit
    system.println('***** findLimit *****');
    system.println('find red dots limit to 2');
    startTimer();
    var list = parent.findLimit(redDot,2);
    stopTimer();
    system.println('found '+list.size());
    system.println('');
    
    // find nonexistent
    system.println('***** find nonexistent image *****');
    system.println('find all yellow dots');
    startTimer();
    var list = parent.findAll(yellowDot);
    stopTimer();
    system.println('found '+list.size());
    system.println('');
    
    // find not optimized images
    system.println('***** find not optimized images *****');
    system.println('find all yellow dots');
    startTimer();
    var list = parent.findAll(notOptimizedYellowDot);
    stopTimer();
    system.println('found '+list.size());
    system.println('');
    
    system.println('find all red dots');
    startTimer();
    var list = parent.findAll(notOptimizedRedDot);
    stopTimer();
    system.println('found '+list.size());
    system.println('');
}
